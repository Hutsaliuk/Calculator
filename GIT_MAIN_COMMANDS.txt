git init		#Создание репозитория в существующем каталоге
git clone <url>		#Клонирование уделённого репозитория

git add *		#Индексирует все файлы в директории
git status		#Определение состояния файлов
git commit -m "comment"	#Фиксация изменений с комментирием
git commit -a -m "comm"	#Индексирует все файлы и фиксирует изминения с комментарием

git commit --amend	#Изменение последнего коммита
git reset HEAD <файл>	#Отменяет индексацию файла
git checkout -- <file>	#Отменить все сделанные изменения в файле

git log			#Просмотр истории коммитов
git log -1		#Посмотреть последний коммит
git log -1 -p		#Посмотреть что изменилось в последнем коммите
git log --stat		#Краткая статистика по каждому коммиту

git diff		#Что изменили, но пока не проиндексировали
git diff --staged	#Разницу между индексированными измененимия и последним коммитом:

git checkout <имя>	#Перейти в другую ветку
git checkout -b <имя>	#Создать ветку и перейти на наё
git merge <имя>		#Сливание указанной ветки с текущей
git branch		#Список всех веток
git branch -v		#Посмотреть последний коммит в каждой ветке
git branch --merged	#Список веток уже слитых с текущей (которые можно удалить)
git branch -d <имя>	#Удалить ветку

git remote add <имя> <url> #Добавление удалённого репозитория
git remote -v		#Отображение удалённых репозиториев
git push origin master  #Отправка ветки в удалённый репозиторий
git fetch origin	#Забирает все обновления с сервера, но не сливает их с моими
git pull origin		#Забирает все обновления с сервера и пытается слить их с моими
git remote show <имя>	#Больше информации об одном из удалённых репозиториев
