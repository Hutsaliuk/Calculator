git init		#Создание репозитория в существующем каталоге

git add *		#Индексирует все файлы в директории
git status		#Определение состояния файлов
git commit -m "comment"	#Фиксация изменений с комментирием

git commit --amend	#Изменение последнего коммита
git reset HEAD <файл>	#Отменяет индексацию файла
git checkout -- <file>	#Отменить все сделанные изменения в файле

git log			#Просмотр истории коммитов
git log -1		#Посмотреть последний коммит
git log -1 -p		#Посмотреть что изменилось в последнем коммите
git log --stat		#Краткая статистика по каждому коммиту

git diff		#что изменили, но пока не проиндексировали
git diff --staged	#разницу между индексированными измененимия и последним коммитом:

git push		#Copy in remote repository
